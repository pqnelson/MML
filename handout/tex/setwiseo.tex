\begin{thm}
\item\label{setwiseo:1} $\{x\}\subset\{x,y,z\}$
\item\label{setwiseo:2} $\{x,y\}\subset\{x,y,z\}$.
\item\label{setwiseo:3} (Cancelled)

\item\label{setwiseo:6} Let $f$ be a function. Then $f(Y\setminus f^{-1}(X))=f(Y)\setminus X$
\end{thm}
